# End-to-end test for the ThunkExample thunk. From the CMake-generated thunk sources (the auto test
# in src/tests/auto/TLC generates them) it builds the guest thunk library (GTL), the host thunk
# library (HTL, with the example host implementation) and a guest program, then the `run_manual_tlc`
# target runs the program under the patched QEMU with the dlcall plugin. Every le_* call the program
# makes is carried across to the host implementation.
#
# It is not part of the normal build or ctest: the targets are EXCLUDE_FROM_ALL and the run target
# is invoked by hand. It only works on an x86_64 host, since the guest side (always x86_64, run under
# qemu-x86_64) reuses the local LoreGuestRT, which on another host is built native.

# Needs the thunk sources generated by the auto test.
if(NOT TARGET LoreTLC OR NOT LORE_TLC_GEN_GUEST_SRC)
    return()
endif()

if(NOT CMAKE_SYSTEM_PROCESSOR MATCHES "x86_64|amd64|AMD64")
    message(WARNING "manual TLC test: skipped on non-x86_64 host '${CMAKE_SYSTEM_PROCESSOR}' "
        "(the guest side reuses the native LoreGuestRT, which cannot serve the x86_64 guest)")
    return()
endif()

set(_fixture ${LORE_TLC_FIXTURE_DIR})
set(_incs ${LORE_SOURCE_DIR}/include ${LORE_BUILD_INCLUDE_DIR} ${_fixture})
set(_work ${CMAKE_CURRENT_BINARY_DIR})

# The sources are produced by a custom command in the auto test's directory; mark them GENERATED
# here so the libraries below can compile them without a configure-time existence check.
set_source_files_properties(${LORE_TLC_GEN_GUEST_SRC} ${LORE_TLC_GEN_HOST_SRC}
    PROPERTIES GENERATED TRUE)

# Common settings for the GTL and HTL: C++20 and no exceptions/rtti. They are deliberately NOT linked
# with --no-undefined, matching the real thunk build: DLCall symbols they do not pull from the
# runtimes are resolved at run time.
function(_te_thunk_lib _target _out)
    set_target_properties(${_target} PROPERTIES
        OUTPUT_NAME ${_out}
        LIBRARY_OUTPUT_DIRECTORY ${_work})
    target_compile_features(${_target} PRIVATE cxx_std_20)
    target_compile_options(${_target} PRIVATE -fno-exceptions -fno-rtti)
    target_include_directories(${_target} PRIVATE ${_incs})
endfunction()

# Guest thunk library (GTL): exports the real le_* symbols the program calls.
add_library(tst_te_gtl SHARED EXCLUDE_FROM_ALL ${LORE_TLC_GEN_GUEST_SRC})
_te_thunk_lib(tst_te_gtl ThunkExample)
target_link_libraries(tst_te_gtl PRIVATE LoreGuestRT)

# Host thunk library (HTL): the generated host thunk plus the example host implementation.
add_library(tst_te_htl SHARED EXCLUDE_FROM_ALL
    ${LORE_TLC_GEN_HOST_SRC} ${_fixture}/ThunkExample.cpp)
_te_thunk_lib(tst_te_htl ThunkExample_HTL)
target_link_libraries(tst_te_htl PRIVATE LoreHostRT)

# Guest program: an ordinary x86_64 program calling the le_* API through the GTL.
add_executable(tst_te_program EXCLUDE_FROM_ALL Program.c)
set_target_properties(tst_te_program PROPERTIES
    OUTPUT_NAME Program
    RUNTIME_OUTPUT_DIRECTORY ${_work})
target_include_directories(tst_te_program PRIVATE ${_fixture})
target_link_libraries(tst_te_program PRIVATE tst_te_gtl)

# The thunk database the host runtime reads: one forward thunk, resolved to the work directory by
# the GTL_DIR / HTL_DIR config variables passed at run time.
file(WRITE ${_work}/ThunkDB.json
    "{\n    \"forwardThunks\": [\n        \"libThunkExample\"\n    ],\n    \"reversedThunks\": []\n}\n")

# The QEMU build providing qemu-x86_64 and the dlcall plugin. This is the only thing the test needs
# pointed at your machine, and it has no default: give it either through the QEMU_BUILD_DIR
# environment variable when the target runs (no reconfigure),
#   QEMU_BUILD_DIR=/path/to/qemu/build/release cmake --build <build> --target run_manual_tlc
# or at configure time with -DQEMU_BUILD_DIR=...
set(QEMU_BUILD_DIR "" CACHE PATH
    "QEMU build directory with qemu-x86_64 and contrib/plugins/libdlcall.so")

# Run the program under QEMU. The actual launch lives in Run.cmake, run via `cmake -P` so it
# can read QEMU_BUILD_DIR from the environment at build time (a custom-target command would bake the
# value in at configure time). The build-time bits (program, work dir, runtime dir, the cache
# QEMU_BUILD_DIR) are passed as -D defines; the generator expressions resolve when the target runs.
add_custom_target(run_manual_tlc
    COMMAND ${CMAKE_COMMAND}
        -D PROGRAM=$<TARGET_FILE:tst_te_program>
        -D WORK=${_work}
        -D RT_DIR=$<TARGET_FILE_DIR:LoreGuestRT>
        -D QEMU_BUILD_DIR=${QEMU_BUILD_DIR}
        -P ${CMAKE_CURRENT_SOURCE_DIR}/Run.cmake
    USES_TERMINAL
    VERBATIM
    COMMENT "Running ThunkExample under QEMU"
)
add_dependencies(run_manual_tlc tst_te_gtl tst_te_htl tst_te_program)
